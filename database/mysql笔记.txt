//mysql.h
//连接数据库
mysql_real_connect()
//设置字符集
mysql_set_character_set()
//mysql默认超过8小时没有使用断开连接，设置自动重连，保持长连接
mysql_options( &mysql, MYSQL_OPT_RECONNECT, … )
//关闭数据库连接
mysql_close()
//数据库查询
mysql_real_query()
//返回数据库错误信息
mysql_errono()
//返回全部查询结果
mysql_store_result()
//需要用时再从服务器读取
mysql_use_result()
//转义 SQL 语句中使用的字符串中的特殊字符
mysql_real_escape_string()
//返回前一次 MySQL 操作所影响的记录行数
mysql_affected_rows()
//返回上一步 INSERT 操作产生的 ID
mysql_insert_id()
